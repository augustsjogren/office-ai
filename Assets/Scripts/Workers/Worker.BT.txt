
		

tree "Root"
	fallback
		tree("Break")
		tree("Lunch")
		tree("FulfillNeeds")
		tree("Work")


tree "FulfillNeeds"
	sequence
		GoToBathroom()
		GetCoffee()
		GetSnack()
		GetLunch()
		
tree "Idle"
	StandStill()

tree "Work"
	Work()

tree "Break"
	while IsBreak
		sequence
			fallback
				GetCoffee()
				GetSnack()
			Work()

tree "Lunch"
	while IsLunch
		sequence
			GetLunch()
			Work()
