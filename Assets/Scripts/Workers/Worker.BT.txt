tree "Root"
	fallback
		tree("VisitBathroom")
		tree("GetDrink")
		tree("GetLunch")
		tree("GetSnack")
		tree("Work")

tree "VisitBathroom"
	sequence
		while NeedToPee
			GoToBathroom()
		Wee()
		GoToSink()
		Wait(2.0)
		WashHands()


tree "GetDrink"
	while IsThirsty
		fallback
			tree("GetCoffee")
			tree ("GetTea")


tree "GetCoffee"
		sequence
			GoToCoffeeMachine()
			Drink()

tree "GetTea"
		sequence
			GoToTeaMachine()
			Drink()

tree "GetSnack"
	while IsHungry
		sequence
			GoToSnackMachine()
			EatSnack()

tree "GetLunch"
	while IsHungry
		sequence
			GoToRestaurant()
			Wait(5.0)
			EatLunch()
		
tree "Idle"
	StandStill()

tree "Work"
	sequence
		GoToDesk()
		Work()
