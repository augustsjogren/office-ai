tree "Root"
	fallback
		tree("VisitBathroom")
		tree("GetCoffee")
		tree("GetLunch")
		tree("GetSnack")
		tree("Work")

tree "VisitBathroom"
	while NeedToPee
		sequence
			GoToBathroom()
			Wee()

tree "GetCoffee"
	while IsThirsty
		sequence
			GoToCoffeeMachine()
			DrinkCoffee()

tree "GetSnack"
	while IsHungry
		sequence
			GoToSnackMachine()
			EatSnack()

tree "GetLunch"
	while IsHungry
		sequence
			GoToRestaurant()
			EatLunch()
		
tree "Idle"
	StandStill()

tree "Work"
	Work()
